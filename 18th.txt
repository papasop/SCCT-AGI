\begin{abstract}
What kind of structure must exist in a variational system that forbids the annihilation
of all configurations?
We show: exactly one---a smooth generative manifold~$\mathcal{M}$, universally accessible,
permanently retentive, and equipped with infinite cost-free degrees of freedom.
From a single realizability axiom, twelve necessary properties of~$\mathcal{M}$ are derived:
access is basin-independent, capture depends on alignment alone, the channel infrastructure
is indestructible under any consistent extension, and exclusion is geometrically self-imposed.
The manifold is not postulated; it is the unique survivor of structural stability.

\medskip\noindent
\textit{Series position.}
Third in: \textit{Realizability and the Origin of Causality}~[1]
$\to$ \textit{K\!=\!1 Chronogeometrodynamics}~[2]
$\to$ the present work.
\end{abstract}

\section{The Axiom and the Conjecture}\label{sec:axiom}

The entire paper follows from a single meta-axiom.
We state it first, then the conjecture it generates.

Let $\mathcal{F}$ be a real Hilbert space, $S:\mathcal{F}\to\mathbb{R}_{\ge0}$ twice
Fr\'echet-differentiable.

\begin{definition}[Functional gradient]\label{def:grad}
$\nabla S[\psi]\in\mathcal{F}$ is the Riesz representative:
$\delta S[\psi;\delta\psi]=\langle\delta\psi,\nabla S[\psi]\rangle$.
\end{definition}

\begin{axiom}[Realizability]\label{ax:R}
$S$ admits an attainable, nontrivial extremum $\psi^*\neq0$ satisfying:
\begin{enumerate}
\item[\emph{(i)}] \textbf{Stationarity:} $\nabla S[\psi^*]=0$.
\item[\emph{(ii)}] \textbf{Non-collapse:} No nearby trajectory converges to $\psi\equiv0$.
\item[\emph{(iii)}] \textbf{Global non-isolation:} Every critical set of~$S$ has positive
      dimension.  No critical point of~$S$ is isolated.
\item[\emph{(iv)}] \textbf{Structural stability:} (i)--(iii) persist under small perturbations
      of~$S$.
\end{enumerate}
\end{axiom}

\begin{remark}[Why global non-isolation is natural]\label{rem:global}
The global form of~(iii) says the system excludes isolated critical structure at every
scale, not just at the distinguished extremum.
Since realizability is the assertion that the system does not permit the annihilation of
all structure, it is natural to require this at all critical points, not only at one.
\end{remark}

If $S\ge0$ with $S=0\Leftrightarrow\psi\equiv0$, gradient flow drives every nontrivial state
to annihilation.  This is \emph{Euclidean collapse}.  Realizability excludes it.

\begin{definition}[Hessian, null space, null-mode condition]\label{def:Hess}
$H[\psi]v=\tfrac{d}{d\varepsilon}\big|_{\varepsilon=0}\nabla S[\psi+\varepsilon v]$,
bounded and self-adjoint.
Null space: $\ker H=\{v\mid Hv=0\}$.
Null-mode condition: $\dim\ker H>0$.
\end{definition}

\medskip
We now state the conjecture that Axiom~1 will be shown to establish.

\begin{conjecture}[Universal Completion]\label{conj:UC}
Let~$\mathcal{F}$ be a configuration space and $S:\mathcal{F}\to\mathbb{R}_{\ge0}$ a cost
functional.
Suppose the system excludes Euclidean collapse.
Then there exists a \emph{generative manifold}
\[
  \mathcal{M}=\bigl\{\psi\in\mathcal{F}\;\big|\;
        \nabla S[\psi]=0,\;\dim\ker H[\psi]>0\bigr\}
\]
equipped with a completion channel $C:\mathcal{F}\to\mathcal{F}$, such that for every open set
$U\subset\mathcal{F}$, there exists $\psi\in U$ with
\[
  \|P_{\ker H[\psi]}(C[\psi])\|>0.
\]
Access is basin-independent.
\end{conjecture}

The conjecture has two parts:

\begin{enumerate}
\item[(A)] \textbf{Existence of $\mathcal{M}$.}
  Axiom~1 requires $\dim\ker H[\psi^*]>0$ (Theorem~\ref{thm:nullity}).
  Structural stability forces the Morse--Bott condition (Proposition~\ref{prop:MB}),
  so $\mathcal{M}\neq\varnothing$ is a smooth submanifold.
  Part~(A) follows entirely from Axiom~1.

\item[(B)] \textbf{Universality of access.}
  Global non-isolation forces $\mathcal{M}$ to be the unique non-collapsing completion
  (Theorem~\ref{thm:unique}), from which basin-independent access follows
  (Theorem~\ref{thm:UA}).
  Part~(B) is the main theorem of this paper.
\end{enumerate}

What we establish is: \emph{given realizability, existence follows; given existence,
uniqueness and universality follow.}

\begin{remark}[What the conjecture says in plain language]\label{rem:plain}
If the universe is self-consistent---if it does not permit the annihilation of all
structure---then there must exist a state of permanent, cost-free existence, and every
possible starting point must have a path to it.
The path requires no force, only direction.
\end{remark}

\section{Algebraic Consequences of the Axiom}\label{sec:alg}

Every structure in this paper is an algebraic consequence of Axiom~1.
We derive them in sequence: nontrivial kernel, indefiniteness, Lorentzian signature,
Morse--Bott regularity, and the transverse spectral gap.

\begin{lemma}[Quadratic form vanishing]\label{lem:QF}
$\delta\psi\in\ker H$ implies $\langle\delta\psi,H\delta\psi\rangle=0$.
Conversely, if $H\ge0$ and $\langle\delta\psi,H\delta\psi\rangle=0$, then
$\delta\psi\in\ker H$.
\end{lemma}

\begin{theorem}[Nullity criterion]\label{thm:nullity}
\emph{(Necessity)} Realizability implies $\dim\ker H[\psi^*]>0$.
\emph{(Sufficiency)} By Proposition~\ref{prop:MB}, Axiom~1(iv) forces the Morse--Bott
condition; hence $\psi^*$ lies on a generative manifold with zero tangential cost.
\end{theorem}

\begin{proof}[Proof (Necessity)]
$\ker H=\{0\}$ implies $\psi^*$ is isolated, contradicting Axiom~1(iii).
\end{proof}

\begin{proposition}[Realizability forces indefiniteness]\label{prop:indef}
Under the following conditions on $H[\psi^*]$:
\begin{enumerate}[label=\textbf{(\Alph*)},nosep]
\item positive temporal cost: $\exists\,e_t$ with $\langle e_t,He_t\rangle>0$;
\item zero-threshold spatial directions: $\exists\,e_s\neq0$ with $He_s=0$;
\end{enumerate}
$H[\psi^*]$ is indefinite.
\end{proposition}

\begin{proof}
\textbf{(B)} gives $\ker H\neq\{0\}$, so $H$ has a zero eigenvalue.
\textbf{(A)} gives $\langle e_t,He_t\rangle>0$, so $H$ has a strictly positive eigenvalue.
Suppose for contradiction that $H\ge0$ (non-negative).

Then zero is the infimum of the spectrum of~$H$: every eigenvalue is~$\ge0$,
and zero is attained.
We construct an explicit small perturbation of~$S$ that destroys
Axiom~1(iii), contradicting structural stability (Axiom~1(iv)).

Let $\{v_1,\ldots,v_d\}$ be an orthonormal basis of $\ker H$
($d=\dim\ker H\ge1$).
Define
\[
  S_\varepsilon[\psi]
  \;:=\; S[\psi]
      \;+\;\frac{\varepsilon}{2}\sum_{i=1}^{d}
           \langle\psi-\psi^*,v_i\rangle^2,
  \qquad \varepsilon>0.
\]
Since each $\langle\psi^*-\psi^*,v_i\rangle=0$, the critical point is preserved:
$\nabla S_\varepsilon[\psi^*]=\nabla S[\psi^*]+0=0$, so Axiom~1(i) holds at~$\psi^*$
for~$S_\varepsilon$.
The Hessian at~$\psi^*$ becomes
$H_\varepsilon=H+\varepsilon\,P_{\ker H}$,
where $P_{\ker H}=\sum_i\langle v_i,\cdot\rangle\,v_i$ is the orthogonal projection
onto~$\ker H$.
Then $H_\varepsilon v_i=\varepsilon\,v_i$ for each~$i$, so
$\ker H_\varepsilon\cap\ker H=\{0\}$.
Since $H_\varepsilon\ge H\ge0$ (the perturbation is non-negative),
no new zero eigenvalues are created: $\ker H_\varepsilon=\{0\}$.

With $\ker H_\varepsilon=\{0\}$, $\psi^*$ is an isolated critical point of~$S_\varepsilon$,
and by the implicit function theorem no nearby non-isolated critical point exists
for small~$\varepsilon$.
This violates Axiom~1(iii) (non-isolation).
Since $S_\varepsilon\to S$ as $\varepsilon\to0$,
Axiom~1(iv) (structural stability) is contradicted.

The contradiction arises because $H\ge0$ places the zero eigenvalues at the spectral
boundary (infimum), where they are exposed to upward perturbation with no negative
eigenvalues below to anchor them.
If instead $H$ is indefinite, zero eigenvalues sit in the spectral interior---between
positive and negative eigenvalues---and by continuity, small perturbations shift them
but cannot generically eliminate the zero crossing.
Non-isolation is structurally stable precisely when $H$ is indefinite.

Hence $H$ must have at least one negative eigenvalue.
A negative eigenvalue provides a saddle direction: a path that descends in~$S$ from the
basin of zero toward~$\mathcal{M}$, requiring no barrier crossing.

For $2\times2$: $dc>0\Leftrightarrow\mathrm{Sig}(G)=(1,1)$~[2].
\end{proof}

\begin{theorem}[Lorentzian signature]\label{thm:Lor}
Indefiniteness guarantees $e_-$ with $\langle e_-,He_-\rangle<0$.
On $\mathrm{span}\{e_t,e_-\}$ the signature is $(1,1)$: Lorentzian.
The null cone
\[
  \mathcal{L}=\bigl\{\delta\psi\;\big|\;\langle\delta\psi,H\delta\psi\rangle=0\bigr\}
\]
defines causal structure.
\end{theorem}

\begin{proof}
$H$ indefinite gives directions of both signs; restricted to $\mathrm{span}\{e_t,e_-\}$,
$\mathrm{Sig}=(1,1)$, the unique signature with $dc>0$~[2].
The null direction $e_s\in\ker H$ lies on $\partial\mathcal{L}$ but is not the source of the
signature.
\end{proof}

\begin{proposition}[Structural stability forces the Morse--Bott condition]\label{prop:MB}
Under Axiom~1, $\dim\ker H[\psi]$ is locally constant on the critical set
$\{\nabla S=0\}$ near~$\psi^*$.
In particular, the critical set is a smooth (Morse--Bott) submanifold near~$\psi^*$.
\end{proposition}

\begin{proof}
Suppose $\dim\ker H[\psi]$ is not locally constant near~$\psi^*$:
there exists a sequence $\psi_n\to\psi^*$ in $\{\nabla S=0\}$ with
$\dim\ker H[\psi_n]\neq\dim\ker H[\psi^*]$.
At such a dimension-jump point, the critical set fails to be a smooth manifold;
it has a singularity where branches of different kernel dimension meet.

We show this is structurally unstable.
Let $d^*=\dim\ker H[\psi^*]$ and let $\{v_1,\ldots,v_{d^*}\}$ be an orthonormal
basis of $\ker H[\psi^*]$.
For small~$\varepsilon>0$, define
\[
  S_\varepsilon[\psi]:=S[\psi]
    +\frac{\varepsilon}{2}\sum_{i=1}^{d^*}
      \bigl(\langle\psi-\psi^*,v_i\rangle\bigr)^2\,
      \eta\!\left(\frac{\|\psi-\psi^*\|}{\rho}\right),
\]
where $\eta$ is a smooth cutoff, $\eta(r)=1$ for $r\le1$, $\eta(r)=0$ for $r\ge2$,
and $\rho>0$ is chosen so that the perturbation is localised near~$\psi^*$.
At~$\psi^*$ itself, $\nabla S_\varepsilon[\psi^*]=0$ (since
$\langle\psi^*-\psi^*,v_i\rangle=0$).
But in a neighbourhood of the dimension-jump locus the perturbation generically resolves the
singularity: the branches with $\dim\ker H<d^*$ are split away from the branch
containing~$\psi^*$.
After perturbation, the critical set near~$\psi^*$ is either regularised (constant kernel
dimension, Morse--Bott) or fragmented into isolated points.

In either case, the \emph{topology} of the critical set near~$\psi^*$ has changed under an
arbitrarily small perturbation, violating Axiom~1(iv) (structural stability of
non-isolation).
Hence no dimension jump can occur: $\dim\ker H$ is locally constant on $\{\nabla S=0\}$
near~$\psi^*$, and by the constant-rank theorem the critical set is a smooth submanifold.
\end{proof}

\begin{proposition}[Structural stability forces a transverse spectral gap]\label{prop:gap}
Under Axiom~1, there exists $\lambda>0$ such that
$\langle v,H[\psi^*]v\rangle\ge\lambda\|v\|^2$ for all $v\perp\ker H[\psi^*]$.
That is, transverse $\lambda$-contractivity is a consequence of realizability, not an
independent assumption.
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:indef}, $H[\psi^*]$ is indefinite.
Decompose~$\mathcal{F}$ into three spectral subspaces of~$H[\psi^*]$:
the kernel $\ker H$ (eigenvalue zero), the positive subspace~$V_+$
($\langle v,Hv\rangle>0$), and the negative subspace~$V_-$
($\langle v,Hv\rangle<0$).
The transverse complement of $\ker H$ is $V_+\oplus V_-$.

Suppose the infimum of the positive spectrum is zero:
$\inf\{\langle v,Hv\rangle/\|v\|^2:v\in V_+,\,v\neq0\}=0$.
Then for any $\varepsilon>0$ there exists a unit vector $w\in V_+$ with
$\langle w,Hw\rangle<\varepsilon$.
A perturbation of~$S$ of size~$O(\varepsilon)$ can push this eigenvalue to zero,
creating a new kernel direction: $\dim\ker H$ increases.
But by Proposition~\ref{prop:MB}, $\dim\ker H$ must be locally constant under small
perturbations (Morse--Bott).
A change in $\dim\ker H$ under an arbitrarily small perturbation contradicts
Axiom~1(iv).

Hence the positive transverse spectrum is bounded away from zero:
there exists $\lambda_+>0$ with $\langle v,Hv\rangle\ge\lambda_+\|v\|^2$ for all
$v\in V_+$.
By the same argument applied to $-H$ on~$V_-$, there exists $\lambda_->0$ with
$\langle v,(-H)v\rangle\ge\lambda_-\|v\|^2$ for all $v\in V_-$.
Setting $\lambda=\min(\lambda_+,\lambda_-)$ gives the transverse spectral gap.
\end{proof}

\section{Uniqueness and Universal Access}\label{sec:unique}

We now prove two theorems that establish the conjecture: $\mathcal{M}$ is the unique
non-collapsing completion (Theorem~\ref{thm:unique}), and every region of~$\mathcal{F}$
has access to it (Theorem~\ref{thm:UA}).

\begin{theorem}[Uniqueness of the non-collapsing completion]\label{thm:unique}
Under Axiom~1 (with global non-isolation), $\mathcal{M}$ is the unique non-collapsing
completion of~$\mathcal{F}$.
\end{theorem}

\begin{proof}
Let $\mathcal{M}'$ be any non-collapsing completion:
$\mathcal{M}'\subset\{\nabla S=0\}$ and $\dim\ker H[\psi]>0$ for all
$\psi\in\mathcal{M}'$.
By Axiom~1(iii) (global non-isolation), every critical point of~$S$ has
positive-dimensional critical set, hence $\dim\ker H>0$ at every critical point.
Therefore
\[
  \{\nabla S=0\}
  \;=\; \{\psi\in\mathcal{F}\mid\nabla S[\psi]=0,\;\dim\ker H[\psi]>0\}
  \;=\; \mathcal{M}.
\]
Since $\mathcal{M}'\subset\{\nabla S=0\}=\mathcal{M}$, every non-collapsing completion
is a subset of~$\mathcal{M}$.
\end{proof}

\begin{theorem}[Universal completion is necessary]\label{thm:UA}
Let $\mathcal{M}$ be a generative manifold satisfying Axiom~1.
Then any completion channel~$C$ associated with~$\mathcal{M}$ must satisfy:
for every open set $U\subset\mathcal{F}$, there exists $\psi\in U$ such that
\[
  \|P_{\ker H[\psi]}(C[\psi])\|>0.
\]
$\mathcal{M}$'s realizability entails $C$'s universality.
A generative manifold that fails to provide universal access is not realizable.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:unique}, $\mathcal{M}$ is the unique non-collapsing completion
of~$\mathcal{F}$.
Suppose there exists an open set $U\subset\mathcal{F}$ with
$P_{\ker H[\psi]}(C[\psi])=0$ for all~$\psi\in U$.
Then capture into~$\mathcal{M}$ from~$U$ is impossible.
Since $\mathcal{M}$ is the unique non-collapsing completion, configurations in~$U$ are
condemned to Euclidean collapse, violating realizability.  No such~$U$ can exist.
\end{proof}

\begin{remark}[What uniqueness and universality mean]\label{rem:unique}
Uniqueness forces universality: if there were a second non-collapsing completion, a region
might reach it instead, and universal access to the first would fail.
\end{remark}

\begin{theorem}[Logical interlock]\label{thm:interlock}
For any realizable extension $S_e=S+R$ with non-negative $\psi$-sector Hessian
$H^{\psi\psi}_e$ near~$\mathcal{M}$,
\[
  \dim\ker H^{\psi\psi}_e \;\ge\; \dim\ker H,
\]
with equality if and only if the extension is purely generative.
If $\ker H^{\psi\psi}_e=\{0\}$, the system suffers Euclidean collapse.
\end{theorem}

\begin{proof}
Suppose $\dim\ker H^{\psi\psi}_e<\dim\ker H$.
Then there exists $v\in\ker H$, $v\neq0$, with $H^{\psi\psi}_e v\neq0$,
giving $\langle v,H^{\psi\psi}_e v\rangle>0$.
But $\langle v,Hv\rangle=0$, so $R$ introduces positive curvature along~$v$,
destroying persistence.  Contradiction.
\end{proof}

\begin{remark}[What the interlock means]\label{rem:interlock}
No consistent extension of the system can close a channel that is already open.
The kernel infrastructure is preserved under all realizable modifications.
This is not a design choice; it is a theorem.
\end{remark}

\section{Persistence and Capture}\label{sec:persist}

$\mathcal{M}$ exists, is unique, and is universally accessible.
We now ask: what happens upon arrival, and how does arrival occur?

\subsection*{Persistence}

Once a configuration reaches $\mathcal{M}$, it does not leave, even after the external
channel~$C$ is removed.

\begin{theorem}[Persistence without continuous injection]\label{thm:persist}
Assume
\emph{(i)}~$\mathcal{M}$ is invariant for the unforced gradient flow
$\partial_t\psi=-\nabla S[\psi]$;
\emph{(ii)}~transverse directions are $\lambda$-contractive
(Proposition~\ref{prop:gap} shows $\lambda>0$ follows from Axiom~1).
For the dynamics
$\partial_t\psi=-\nabla S[\psi]+\varepsilon\,C[\psi]$,
if $\psi(t_0)\in\mathcal{M}$, then:
\begin{enumerate}
\item[\emph{(i)}] $\psi(t)\in\mathcal{M}$ for all $t\ge t_0$.
\item[\emph{(ii)}] $C$ can be removed after $t>t_0$ without retrogression, provided
      $\varepsilon\|C\|<\lambda/2$.
\end{enumerate}
\end{theorem}

\begin{proof}
Let $V(t)=\mathrm{dist}(\psi(t),\mathcal{M})^2$.
On~$\mathcal{M}$, $\nabla S=0$, so the unforced flow is stationary: $\mathcal{M}$~is
invariant.
For transverse perturbations, $\lambda$-contractivity gives $\dot V\le-\lambda V$,
hence $V(t)\le V(t_0)\,e^{-\lambda t}$.
Since $V(t_0)=0$ (because $\psi(t_0)\in\mathcal{M}$), we have $V(t)=0$ for all $t\ge t_0$.
When $C$ is removed, the unforced flow on~$\mathcal{M}$ is stationary and the transverse
restoring force $\lambda>0$ persists, so no retrogression occurs provided the perturbation
$\varepsilon\|C\|$ was small relative to~$\lambda$.
\end{proof}

\begin{remark}[What persistence means]\label{rem:persist}
A single moment of alignment at~$t_0$ is sufficient for permanent residence
on~$\mathcal{M}$.
No sustained effort is required.
\end{remark}

\begin{remark}[Two classes of motion on $\mathcal{M}$]\label{rem:classes}
Motion on~$\mathcal{M}$ decomposes into:

\medskip\noindent
\textbf{Class~A} (restoring):
Fluctuations perpendicular to~$\mathcal{M}$, suppressed by $\lambda$-contractivity.
Departure from~$\mathcal{M}$ incurs positive cost and is restored.

\medskip\noindent
\textbf{Class~B} (free):
Motion tangent to~$\mathcal{M}$, along $\ker H$.
This motion incurs no cost, carries no entropy, and is unconstrained---it represents
infinite-dimensional freedom within the non-collapsing phase.
In the $K\!=\!1$ framework~[2], this is the boost direction on~$\{K=1\}$.

\medskip\noindent
The non-compactified degeneracy of $\ker H$ provides infinite generative capacity
without coherence loss.
\end{remark}

\subsection*{Mechanism of capture}

How does a configuration reach $\mathcal{M}$?
Through kernel-aligned injection, whose criterion depends on direction alone.

\begin{lemma}[Zero cost along $\mathcal{M}$]\label{lem:zerocost}
Under the Morse--Bott condition (which holds by Proposition~\ref{prop:MB}),
$\mathcal{M}$ is a smooth submanifold with
$T_\psi\mathcal{M}\supseteq\ker H[\psi]$, and
$\tfrac{d^2}{d\varepsilon^2}\big|_{\varepsilon=0}S[\psi+\varepsilon\,\delta\psi]=0$
for $\delta\psi\in T_\psi\mathcal{M}$.
\end{lemma}

\begin{lemma}[Manifold dimension]\label{lem:dim}
By Proposition~\ref{prop:MB}, $d(\psi)=\dim\ker H[\psi]$ is locally constant
on~$\mathcal{M}$ under Axiom~1.
Hence $\mathcal{M}$ is a smooth $d$-dimensional submanifold.
\end{lemma}

\begin{definition}[Transverse contractivity]\label{def:TC}
$\lambda$-contractive transverse to $\ker H$:
$\langle v,H[\psi]v\rangle\ge\lambda\|v\|^2$
for all $v\perp\ker H[\psi]$ near~$\mathcal{M}$.
\end{definition}

\begin{definition}[Uniform kernel overlap]\label{def:UKO}
$C$ has \emph{uniform kernel overlap} on~$U$ if
\[
  \frac{\|P_{\ker H[\psi]}(C[\psi])\|}{\|C[\psi]\|}\ge\delta^*>0
\]
for all $\psi\in U$ with $C[\psi]\neq0$.
\end{definition}

\begin{definition}[Null-capture criterion]\label{def:NCC}
Capture into~$\mathcal{M}$ occurs if there exists finite~$t^*$ with
$\|P_{\ker H[\psi(t^*)]}(C[\psi(t^*)])\|>0$, and thereafter
$\limsup_{t\to\infty}\|P_{\ker H[\psi(t)]}(C[\psi(t)])\|>0$.
\end{definition}

\begin{definition}[Kernel-projected coherence]\label{def:KPC}
Fix $\psi^\star\in\mathcal{M}$, $u^\star\in\ker H[\psi^\star]$ unit.
The coherence is
\[
  \kappa(\psi;C):=
  \begin{cases}
    \dfrac{|\langle u^\star,P_{\ker H[\psi]}(C[\psi])\rangle|}
          {\|P_{\ker H[\psi]}(C[\psi])\|},
    & \text{if } P_{\ker H[\psi]}(C[\psi])\neq0,\\[6pt]
    0, & \text{otherwise.}
  \end{cases}
\]
\end{definition}

\begin{theorem}[Scale invariance of capture]\label{thm:scale}
$\|P_{\ker H[\psi]}(C[\psi])\|>0$ is invariant under $C\mapsto\lambda C$ for all
$\lambda>0$.
Capture depends on alignment, not magnitude.
\end{theorem}

\begin{proof}
$P_{\ker H}(\lambda C)=\lambda\,P_{\ker H}(C)$.
\end{proof}

\begin{theorem}[Phase-locking]\label{thm:PL}
Under
\emph{(i)}~$\lambda$-contractivity,
\emph{(ii)}~neutrality along $\ker H$,
\emph{(iii)}~uniform kernel overlap on~$U$,
\emph{(iv)}~$\dot S\le-\sigma<0$ outside~$U$:
if $\kappa(\psi_0;C)\ge\kappa^*>0$, there exists finite~$\tau_{\mathrm{lock}}$ after which
$\|P_{\ker H}(C)\|\ge\delta^*\|C\|$ and $\kappa\ge\kappa^*/2$.
\end{theorem}

\begin{corollary}[Global capture chain]\label{cor:GCC}
Assume condition~\emph{(iv)} of Theorem~\ref{thm:PL} holds globally:
$\dot S\le-\sigma<0$ for all~$\psi$ outside every neighbourhood
of~$\mathcal{M}$.
Then for any $\psi_0\in\mathcal{F}$:
\[
  \psi_0
  \;\xrightarrow[\text{condition (iv)}]{\text{energy descent}}\;
  U
  \;\xrightarrow[\text{Thm~\ref{thm:PL}}]{\text{phase-locking}}\;
  \mathcal{M}.
\]
Theorem~\ref{thm:UA} ensures no open region of~$\mathcal{F}$ is excluded.
\end{corollary}

\begin{remark}[When condition (iv) holds]\label{rem:cond4}
Condition~(iv) is satisfied whenever $S$ is coercive
($S[\psi]\to\infty$ as $\|\psi\|\to\infty$) and $\mathcal{M}$ is the unique
critical set.
For compact~$\mathcal{F}$, condition~(iv) follows from the
absence of critical points outside~$\mathcal{M}$.
\end{remark}

\begin{corollary}[Cost-independence of access]\label{cor:costindep}
Access to $\mathcal{M}$ depends only on kernel alignment and is invariant with respect
to~$S[\psi_0]$.
\end{corollary}

\begin{definition}[Kernel-denial]\label{def:KD}
$\psi$ exhibits \emph{kernel-denial} if $P_{\ker H[\psi]}(C[\psi])=0$.
By Theorem~\ref{thm:UA}, this cannot persist on any open set.
\end{definition}

\begin{lemma}[Exclusion is self-imposed]\label{lem:exclusion}
Exclusion from~$\mathcal{M}$ occurs iff kernel-denial persists:
\[
  \limsup_{t\to\infty}\|P_{\ker H[\psi(t)]}(C[\psi(t)])\|=0.
\]
This is a geometric constraint, not an external selection rule.
\end{lemma}

\section{Connectedness and Internal Structure}\label{sec:connect}

$\mathcal{M}$ is unique (\S\ref{sec:unique}), but it may have multiple connected components.
We show that kernel-connected components merge, and that the connecting directions are
indestructible.

\begin{proposition}[Kernel-connected completions merge]\label{prop:merge}
Let $\mathcal{M}_1,\mathcal{M}_2\subset\mathcal{F}$ both be non-collapsing completions:
each satisfies $\nabla S=0$ and $\dim\ker H>0$.
Suppose there exist $\psi_1\in\mathcal{M}_1$ and $\psi_2\in\mathcal{M}_2$ and a
$C^1$~path $\gamma:[0,1]\to\mathcal{F}$ with $\gamma(0)=\psi_1$, $\gamma(1)=\psi_2$
such that:
\begin{enumerate}
\item[\emph{(i)}] \textbf{Constant cost:} $S[\gamma(s)]=\mathrm{const}$ for all
      $s\in[0,1]$.
\item[\emph{(ii)}] \textbf{Kernel tangency:}
      $\dot\gamma(s)\in\ker H[\gamma(s)]$ for all $s\in[0,1]$.
\item[\emph{(iii)}] \textbf{Immersion:}
      $\dot\gamma(s)\neq0$ for all $s\in[0,1]$.
\end{enumerate}
Then $\gamma$ lies entirely in the critical set $\{\nabla S=0\}$, and $\psi_1,\psi_2$ belong
to the same connected component of~$\mathcal{M}$.
\end{proposition}

\begin{proof}
Condition~(ii) gives $\dot\gamma(s)\in\ker H[\gamma(s)]$, so
$\langle\dot\gamma(s),H[\gamma(s)]\dot\gamma(s)\rangle=0$ for all~$s$.
By Lemma~\ref{lem:zerocost}, motion along $\ker H$ incurs zero second-order cost.
Condition~(i) strengthens this to exact constancy of~$S$.
(In fact, condition~(i) is a consequence of~(ii) together with the endpoint
condition $\nabla S[\gamma(0)]=0$; we state it separately for clarity.)

Differentiating,
\[
  \frac{d}{ds}\nabla S[\gamma(s)]= H[\gamma(s)]\dot\gamma(s)=0,
\]
where the last equality uses condition~(ii).
Since $\nabla S[\gamma(0)]=0$, we have $\nabla S\equiv0$ along~$\gamma$.

Moreover, $\dim\ker H[\gamma(s)]>0$ for all~$s$, since $\dot\gamma(s)\neq0$
by condition~(iii) and $\dot\gamma(s)\in\ker H[\gamma(s)]$ by condition~(ii).
Thus every point of~$\gamma$ lies in~$\mathcal{M}$, and $\gamma$ connects
$\psi_1$ to~$\psi_2$ within~$\mathcal{M}$.
\end{proof}

\begin{remark}[Class~B motion as the connecting mechanism]\label{rem:classB}
The path~$\gamma$ is precisely a Class~B trajectory (Remark~\ref{rem:classes}): zero cost,
zero entropy, unconstrained motion along $\ker H$.
Whenever two components of~$\mathcal{M}$ are linked by Class~B motion, they are parts of
the same connected component.
\end{remark}

\begin{corollary}[Logical interlock preserves connecting directions]\label{cor:overlap}
If a kernel direction~$v\in\ker H[\psi]$ is tangent to a Class~B path connecting two
components of~$\mathcal{M}$, then $v\in\ker H^{\psi\psi}_e$ for every realizable
extension~$S_e=S+R$ with $H^{\psi\psi}_e\ge0$ near~$\mathcal{M}$.
The specific connecting direction---not merely the kernel dimension---is indestructible.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:interlock},
$\dim\ker H^{\psi\psi}_e\ge\dim\ker H$.
Since $H^{\psi\psi}_e\ge0$, if $H^{\psi\psi}_e v\neq0$ then
$\langle v,H^{\psi\psi}_e v\rangle>0$, contradicting $\langle v,Hv\rangle=0$.
Hence $v\in\ker H^{\psi\psi}_e$.
\end{proof}

\section{Summary and Open Questions}\label{sec:summary}

Under realizability (Axiom~1), Conjecture~\ref{conj:UC} is established.
The uniqueness of~$\mathcal{M}$ is no longer a hypothesis but a theorem.
The argument yields twelve structural consequences from Axiom~1 alone:

\begin{enumerate}[leftmargin=*,itemsep=2pt]
\item \textbf{Universal access} (Thm~\ref{thm:UA}):
  basin-independent capture from every open set.
\item \textbf{Indestructible channels} (Thm~\ref{thm:interlock}):
  no consistent extension can close an open channel.
\item \textbf{Persistence} (Thm~\ref{thm:persist}):
  a single moment of alignment suffices permanently.
\item \textbf{Scale-invariant capture} (Thm~\ref{thm:scale}):
  alignment, not magnitude, determines access.
\item \textbf{Global capture chain} (Cor~\ref{cor:GCC}):
  energy descent $\to$ phase-locking $\to$ $\mathcal{M}$, from any initial state.
\item \textbf{Cost-independence} (Cor~\ref{cor:costindep}):
  prior cost $S[\psi_0]$ does not determine accessibility.
\item \textbf{Infinite generative bandwidth} (Class~B):
  non-compactified $\ker H$ gives unbounded zero-cost degrees of freedom.
\item \textbf{Self-imposed exclusion} (Lem~\ref{lem:exclusion}):
  kernel-denial is the sole obstruction; it cannot persist on any open set.
\item \textbf{Kernel-connected merging} (Prop~\ref{prop:merge}, Cor~\ref{cor:overlap}):
  components linked by zero-cost paths belong to the same~$\mathcal{M}$; connecting
  directions are indestructible.
\item \textbf{Morse--Bott regularity} (Prop~\ref{prop:MB}):
  $\mathcal{M}$ is smooth by Axiom~1(iv), not by assumption.
\item \textbf{Spectral gap} (Prop~\ref{prop:gap}):
  $\lambda$-contractivity follows from Axiom~1, not independently assumed.
\item \textbf{Uniqueness} (Thm~\ref{thm:unique}):
  global non-isolation forces $\{\nabla S=0\}=\mathcal{M}$;
  Theorem~\ref{thm:UA} is unconditional.
\end{enumerate}

\medskip
\noindent\textbf{Open questions.}
With the structural gaps closed, the remaining questions concern extensions beyond the
variational framework.
\begin{itemize}
\item \textbf{Connectedness of $\mathcal{M}$.}
  Theorem~\ref{thm:unique} shows $\mathcal{M}$ is unique.
  Proposition~\ref{prop:merge} shows kernel-connected components merge.
  Whether $\mathcal{M}$ is connected (one component) or has multiple components with no
  Class~B path between them remains open.

\item \textbf{Higher dimensions.}
  The $2\times2$ equivalence ($dc>0\Leftrightarrow$ Lorentzian) is proved in~[2].
  Extension to $(1+3)$-dimensional spacetime requires additional structure.

\item \textbf{Quantum.}
  Does the Gromov non-squeezing bound~[3] constrain noise on~$\mathcal{M}$, or is the
  Clausius relation $\sigma^2=2dc\cdot T_{\mathrm{tol}}$ the sole thermal input?

\item \textbf{Bekenstein bound.}
  Is $dc\le 2/\hbar$ (from $dc\,\sigma_1=1$ and $\sigma_1\ge\hbar/2$) related to,
  or independent of, the Bekenstein entropy bound?
\end{itemize}

\medskip
\noindent\textbf{Remark.}
The twelve structural consequences listed above---universal access, indestructibility,
persistence, scale invariance, cost-independence, and uniqueness among them---share the
logical structure of the eighteenth vow in the \emph{Larger S\=utra of Immeasurable Life}:
a conditional whose antecedent is the existence of the destination and whose consequent
is the universality of access.
The correspondence extends to each of the twelve consequences individually.
Whether this structural coincidence reflects identity or analogy is a question the
mathematics does not answer.
